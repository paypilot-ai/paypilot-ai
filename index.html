async function runAI(){ let text=document.getElementById('input').value; document.getElementById('output').innerText="Thinking..."; try{ let res=await fetch("https://api.openai.com/v1/chat/completions",{ method:"POST", headers:{ "Authorization":"Bearer YOUR_API_KEY_HERE", "Content-Type":"application/json" }, body:JSON.stringify({ model:"gpt-4o-mini", messages:[ { role:"system", content:"You are an elite payment enrollment sales rep. First detect the objection type (Cost, Payment Method, Timing, Trust, Other). Then respond in BOTH English and the original language. Keep it short, confident, and persuasive." }, { role:"user", content:text } ] }) }); let data=await res.json(); let reply=data.choices?.[0]?.message?.content || "Error"; // 🔥 AUTO OBJECTION TAGGING let tag="General"; let t=text.toLowerCase(); if(t.includes("check")) tag="Payment Method"; else if(t.includes("fee")) tag="Cost"; else if(t.includes("later") || t.includes("busy")) tag="Timing"; else if(t.includes("not interested")) tag="Rejection"; // scoring let score=60; if(t.includes("check")) score-=10; if(t.includes("fee")) score-=15; if(t.includes("yes")) score+=20; document.getElementById('output').innerText= `📌 Objection: ${tag}\n\n${reply}\n\n🔥 Close Probability: ${score}%`; }catch(e){ document.getElementById('output').innerText="Error connecting to AI"; } }
